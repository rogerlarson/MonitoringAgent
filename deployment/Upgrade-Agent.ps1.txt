param(
    [string]$SourceFolder,
    [string]$InstallFolder =
        "C:\MonitoringAgent\Agent"
)

$serviceName =
    "MonitoringAgentAgent"

Write-Host "Stopping service..."

Stop-Service `
    $serviceName `
    -Force

Start-Sleep -Seconds 5

Write-Host "Copying files..."

Copy-Item `
    "$SourceFolder\*" `
    $InstallFolder `
    -Recurse `
    -Force

Write-Host "Starting service..."

Start-Service `
    $serviceName

Write-Host "Upgrade complete."
